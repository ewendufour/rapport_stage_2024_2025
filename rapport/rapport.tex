\section{The Linear Substituton Calculus}
\begin{definition}[Linear Substitution Calculus]
The LSC is given by the following set of terms:
\begin{align*}
    t ::= x~|~t~t~|~\lambda x.t~|~t[x / t]
\end{align*}
together with the reduction rules $\rightarrow_{d\beta},~\rightarrow_{ls}$ and $\rightarrow_{gc}$:
\begin{center}
\begin{tabular}{l l l l }
$(\lambda x.t)Lu$ &$\mapsto_{d\beta}$ &$t[x / u]L$ & \\
$\mathcal{C}\llbracket x \rrbracket [x / u]$ &$\mapsto_{ls}$ &$\mathcal{C}\llbracket u \rrbracket [x / u]$ & \\
$t[x / u]$ &$\mapsto_{gc}$ &$t$ & if $x\notin fv(t)$
\end{tabular}
\end{center}
 where $L$ is a possibly empty list of substitution, $\mathcal{C}$ is a context and $fv(t)$ denotes the free variables of the term $t$.
\end{definition}
\begin{definition}[Free variables]
\begin{align*}
    fv(x) &= \{ x \} \\
    fv(\lambda x.t) &= fv(t) \setminus \{ x \} \\
    fv(t~u) &= fv(t) \cup fv(u) \\
    fv(t[x/u]) &= (fv(t) \setminus \{ x \}) \cup fv(u) \\
\end{align*}
\end{definition}

\begin{definition}[Reduction contexts]
We define the reduction contexts as:
\begin{align*}
    \mathcal{C} ::= \square~|~\mathcal{C}~t~|~v~\mathcal{C}~|~\mathcal{C}[x/t]
\end{align*}

\begin{align*}
    \mathcal{C} ::= \square~|~\mathcal{C}[\square~t]~|~\mathcal{C}[v~\square]~|~\mathcal{C}[\square[x/t]]
\end{align*}
\end{definition}

\begin{definition}[Propreté]
\begin{align*}
    cl(x) &= true\\
    cl(\lambda x.t) &= cl(t)\\
    cl(t~u) &= cl(t) \wedge cl(u)\\
    cl(t[x/u]) &= cl(t) \wedge x\in fv(t) \wedge cl(u)
\end{align*}
Un term est propre si et seulement si $cl(t)=true$.
\end{definition}

\section{Le Space Calculus}

\begin{definition}[Termes du Space Calculus]
Les termes du space calculus sont donnés par la grammaire suivante:
\begin{align*}
    l ::&= x \\
    &|~\lambda x.a \\
    &|~a~a\\
    a ::&= l^{\alpha}\\
    c ::&= a\\
        &|~c[x/c]\\
    t ::&= c\\
        &|~t~c
\end{align*}
\end{definition}
\begin{definition}[Restriction d'un environnement]
\begin{align*}
a_{|t} &= a\\
    (L[x/c])_{|t} &= L_{|t}[x/c] \quad &\text{if } x\in fv(t)\\
    (L[x/c])_{|t} &= L_{|t} \quad &\text{if } x\notin fv(t)
\end{align*}
\end{definition}
