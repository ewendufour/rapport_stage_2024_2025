    \begin{block}{Definition: FaustML}
        A function in FaustML follows the following grammar: 
    \begin{align*}
        \text{ Function } f(x_1,..,x_n) &= e \\
        \text{ Expression } \qquad \quad e ::&= x \\
        & |\; let \; x  =  e_1 \; in \; e_2 \\
        & |\; c(e_1,..,e_n) \\
        & |\; f(e_1,..,e_n) \\
        & |\; match \; e \; with \; (c_i(x_1,..,x_n) \to e_i)_i \\
    \end{align*}
    \end{block}
\end{frame}

\section{data-tiering}
\begin{frame}{Data-tiering}
    \begin{block}{tiering}
        data-tiering is affecting to each expressions and functions integers (called \textbf{tier}) and add constraints to them.
        It is stated as a \textbf{typing} judgment $f \triangleright i_1,..,i_n \to k$
    \end{block} \pause
    for example the composition propagate the tiers:
    \begin{gather*}
        \inference{f \triangleright p \to k \quad g \triangleright i \to p}{f \circ g \triangleright i \to k}
    \qquad \inference{f \text{ recursive } &...& \color{red}{p > k}}{f \triangleright p \to k}
    \end{gather*}
\end{frame}

\section{SIMREC parameter subst}

\begin{frame}{Extension to SIMREC}
    SIMREC structure (from Avanzini and Dal Lago):
\begin{align*}
    f &:= f_c & f() &= c() \label{null-constr0} \tag{0-constr}\\
    f &:= f_c &f(\vec{x}) &= c(\vec{x}) \label{constr0} \tag{constr}\\
    f &:= \Pi_i^n\;\;& f(x_1,..,x_n) &= x_i \label{proj0} \tag{proj}\\
    f &:= h \circ (g_1,..,g_k) &f(\vec{x}) &= h(g_1(\vec{x}),..,g_k(\vec{x})) \label{comp0} \tag{comp}\\
    f &:= case((g_i)_i)& f(c_i(\vec{x}), \vec{y}) &=  g_i(\vec{x}, \vec{y}) \label{case0} \tag{case}  \\
    f_j &:= simrec((h_{ij})_{ij})_j & f_j(c_i(\vec{x}), \vec{y}) &=  h_{ij}(\vec{x}, \vec{y}, f_1(x_1, \vec{y})..f_1(x_n, \vec{y}).. f_k(x_n, \vec{y})) \label{rec0} \tag{rec}
\end{align*}\pause
    \begin{block}{Definition}
        The class \textbf{SAFE} is the smallest class of function containing (0-constr) (proj) and closed under (comp) and (case) 
    \end{block}
\end{frame}
\begin{frame}[fragile]{Safe functions}
    \begin{example}
        \texttt{tl} is a SAFE function:
    \begin{lstlisting}
        tl(l) = match l with 
            | Nil $\to$ Nil 
            | Cons(x,l) $\to$ l
    \end{lstlisting}
    \end{example}    \pause

    \begin{alertblock}{safe function's crutial property}
        if $\pi \in SAFE$ then for any values $\vec{v}$: $STs(\pi(\vec{v})) \subseteq STs(\vec{v}) \cup E$
    \end{alertblock}
    \small STs(v) are the sub-terms of v
\end{frame}
\begin{frame}{PSIMREC}
\begin{align*}
    f &:= f_c &f(\vec{x}) &= c(\vec{x}) \tag{constr}\\
    f &:= \Pi_i^n\;\;& f(x_1,..,x_n) &= x_i  \tag{proj}\\
    f &:= h \circ (g_1,..,g_k) &f(\vec{x}) &= h(g_1(\vec{x}),..,g_k(\vec{x})) \tag{comp}\\
    f &:= case((g_i)_i)& f(c_i(\vec{x}), \vec{y}) &=  g_i(\vec{x}, \vec{y}) \tag{case}  
\end{align*}
\begin{gather*}
    f_j := psimrec((h_{ij})_{ij}, (\vec{\pi}_{ijklzw})_{ijklzw})_j  \tag{p-rec}
\end{gather*}
    \begin{block}{Definition}
        The class \textbf{PSIMREC} is the smallest class of function containing (constr) (proj) and closed under (comp), (case) and (p-rec)
    \end{block}
\end{frame}
\begin{frame}{PSIMREC vs SIMREC}
    $\pi_i$ are SAFE functions
\begin{align*}
    f_i(nil,x,y) &= g_i(x,y) \tag{p-rec}\\
    f_i(bin(z,w),x,y) &= h_i(z,w,x,y,f_1(z,\pi_1(x), \pi_2(y)),f_1(w,\pi_3(x), \pi_4(y)), \\ & \qquad \qquad \qquad \; \;f_1(w,\pi_5(x), \pi_6(y)),f_2(w,\pi_7(x), \pi_8(y)))\\
    \text{-} \\
    f_i(nil,x,y) &= g_i(x,y)\tag{rec}\\
    f_i(bin(z,w),x,y) &= h_i(z,w,x,y,f_1(z,x, y),f_1(w,x, y), \\ & \qquad \qquad \qquad \; \;f_2(z,x, y),f_2(w,x, y))
\end{align*}
\end{frame}
\begin{frame}{PSIMREC Soundness}
    \begin{alertblock}{PSIMREC Soundness}
        For every function $f \in PSIMREC(\mathbb{A})$, with $f \triangleright i_1,..,i_n \to k$.\\
        For all inputs $\vec{v}$
        a DAG representation of $f(\vec{v})$ is \textbf{computable in time polynomial} in $\|\vec{v}\|$
    \end{alertblock}\pause
    % \includegraphics[width=\textwidth]{DAG-tree.png}\tiny *
    *Figure from \textit{Avanzini and Dal Lago "On sharing, Memoization and Polytime" (2018)}
\end{frame}

\begin{frame}{PSIMREC proof}
    For every function $f \in PSIMREC(\mathbb{A})$:
    \begin{block}{poly space}
        it exists $p_f$ such that $\|f(\vec{v})\| \leq p_f(\|\vec{v}\|)$
    \end{block}
    \begin{block}<2->{function call time}
        $(C,f(\vec{v})) \Downarrow_m (D,u)$ with $m=|FC^f_{\vec{v}} \setminus C|$
    \end{block}
    \begin{block}<3->{function call uper bound}
        it exists $q_f$ such that $|FC^f_{\vec{v}}| \leq q_f(\|\vec{v}\|)$ 
    \end{block}
    \small *  $\|\vec{v}\|$ is the maximally shared size of $\vec{v}$ \\
    **$FC^f_{\vec{v}}$ are the function call arising from $f(\vec{v})$
\end{frame}

\begin{frame}{FaustML Soundness}
    \begin{alertblock}{FaustML Soundness}
    For every function $f$ in any version of FaustML above, with $f \triangleright i_1,..,i_n \to k$.\\
        For all inputs $\vec{v}$
        a DAG representation of $f(\vec{v})$ is \textbf{computable in time polynomial} in $\|\vec{v}\|$
     \end{alertblock}\pause
    % \includegraphics[width=\textwidth]{DAG-tree.png}\tiny *
    *Figure from \textit{Avanzini and Dal Lago "On sharing, Memoization and Polytime" (2018)}
\end{frame}

\begin{frame}{FaustML proof}
    $\beta : $ FaustML nested function $\to$ FaustML version of PSIMREC
    \begin{block}{Tiering preservation}
        if $f \triangleright J \to k$ then $\beta(f) \triangleright J \to k$
    \end{block}\pause

    \begin{block}{value correctness}
        if $\emptyset \vdash f(\vec{v}) \downarrow u$ then $\emptyset \vdash \beta(f)(\vec{v}) \downarrow u$
    \end{block} \pause

    \begin{block}{time correctness}
        $FC^{\beta(f)}_{\vec{v}} \subseteq FC^{f}_{\vec{v}}$
    \end{block}
\end{frame}
